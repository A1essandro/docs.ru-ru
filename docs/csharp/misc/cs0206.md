```  
    public static int P  
    {  
        get  
        {  
            return 0;  
        }  
        set

    }

    // public static void MyMeth(int i)

    }

    {  
        MyMeth(ref P);   // CS0206  
        // try the following line instead  
        // MyMeth(P);   // CS0206  
    }  
}  
```
