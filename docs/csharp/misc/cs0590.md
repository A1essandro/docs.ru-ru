```csharp  
{  
   public class a  
   {  
      public static void operator+(a A1, a A2)   // CS0590  
      {  
      }  
  
      // try the following user-defined operator  
      /*  
      public static a operator+(a A1, a A2)  
      {  
         return A2;  
      }  
      */  
  
      public static int Main()  
      {  
         return 1;  
      }  
   }  
}  
```
