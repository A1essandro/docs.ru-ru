## <a name="example"></a><span data-ttu-id="3a379-106">Пример</span><span class="sxs-lookup"><span data-stu-id="3a379-106">Example</span></span>

   public static void M(ref int j)

      j++;

   {  
      M (ref 2);   // CS1510, can't pass a number as a ref parameter

   }  
}
